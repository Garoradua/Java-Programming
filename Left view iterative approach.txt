Leftview(Node root){
    Queue<node> q = new LinkedList<>();
    q.add(root);

    while(!q,isEmpty()){
        int size = q.size();
        Node curr = q.pop();
    for(int i=0; i<size; i++){
        if(i==0) sout(curr.key);
        if(curr.left!=null) q.add(curr.left);
        if(curr.right!=null) q.add(curr.right);
    }
    }
}
